 47:     /**
 48:      * Recusively merges one or more arrays into a 'base' array.
 49:      * @param array $array The base array to merge
 50:      * @return array
 51:      */
 52:     public static function array_merge_recursive(array $array)
 53:     {
 54:         $args = func_get_args();
 55:         $num = func_num_args();
 56:         if ($num == 1)
 57:             return $array;
 58:         
 59:         $merged = $array;
 60:         
 61:         for ($i = 1; $i < $num; $i++) {
 62:             $merge = $args[$i];
 63:             foreach ($merge as $key => $val) {
 64:                 if (is_array($val) && isset($merged[$key]) && is_array($merged[$key])) {
 65:                     $merged[$key] = static::array_merge_recursive($merged[$key], $val);
 66:                 } else {
 67:                     $merged[$key] = $val;
 68:                 }
 69:             }
 70:         }
 71:         
 72:         return $merged;
 73:         
 74:     }
 75:     
 76:     /**
 77:      * Build a URL generated from parse_url() into a string
 78:      * @param array $url An array URL parts (e.g. from parse_url())
 79:      * @param bool $strip_auth
 80:      * @return string
 81:      */
 82:     public static function http_build_url(array $url, $strip_auth = TRUE)
 83:     {
 84:         $url = static::array_merge_recursive([
 85:             'scheme' => 'http',
 86:             'host' => '',
 87:             'port' => 80,
 88:             'user' => '',
 89:             'pass' => '',
 90:             'path' => '',
 91:             'query' => '',
 92:             'fragment' => ''
 93:         ], $url);
 94:         extract($url);
 95:         if (!$strip_auth && !empty($user)) {
 96:             $surl = sprintf('%s://%s@%s', $scheme, (!empty($pass) ? $user.':'.$pass : $user), $host);
 97:         } else {
 98:             $surl = sprintf('%s://%s', $scheme, $host);
 99:         }
100:         if ($port != 80 && $post != 443)
101:             $surl .= ':' . $port;
102:         $surl .= $path;
103:         $surl .= !empty($query) ? '?' . $query : '';
104:         $surl .= !empty($fragment) ? '#' . $fragment : '';
105:         return $surl;
106:     }
