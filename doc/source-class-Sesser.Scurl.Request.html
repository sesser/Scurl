 34: /**
 35:  * Request
 36:  * 
 37:  * The Scurl request object. Call Request::send() to send the request and return
 38:  * the {@link Response Response} object
 39:  *
 40:  * @author Randy Sesser <sesser@gmail.com>
 41:  */
 42: class Request
 43: {
 44:     /** Request method GET */
 45:     const METHOD_GET = 'GET';
 46:     
 47:     /** Request method POST */
 48:     const METHOD_POST = 'POST';
 49:     
 50:     /** Request method PUT */
 51:     const METHOD_PUT = 'PUT';
 52: 
 53:     /** Request method DELETE */
 54:     const METHOD_DELETE = 'DELETE';
 55: 
 56:     /** Request method HEAD */
 57:     const METHOD_HEAD = 'HEAD';
 58: 
 59:     /** @var string The request method */
 60:     public $method = Request::METHOD_GET;
 61:     
 62:     /** @var string The URL for this request */
 63:     public $url = '';
 64:     
 65:     /** @var array The request parameters */
 66:     public $parameters = [];
 67:     
 68:     /** @var mixed Generally used for PUT requests */
 69:     public $data = NULL;
 70:     
 71:     /** @var array The headers sent to the server */
 72:     public $headers = [];
 73:     
 74:     /** @var array cURL options to set */
 75:     public $options = [];
 76:     
 77:     /** @var array Authentication parameters [ 'user' => (string), 'pass' => (string)] */
 78:     protected $auth = [];
 79:     
 80:     /** @var array Configuration for this request */
 81:     protected $config = [];
 82:     
 83:     /** @var array Curl options container */
 84:     private $curlopts = [];
 85:     
 86:     /**
 87:      * Constructs a new Request object
 88:      * @param array $config Configuration for this request
 89:      */
 90:     public function __construct($url = '', array $config = [])
 91:     {
 92:         $this->url = $url;
 93:         $this->config = $config;
 94:         
 95:         $this->initialize([
 96:             'method' => Request::METHOD_GET,
 97:             'auth' => [
 98:                 'user' => '',
 99:                 'pass' => ''
100:             ],
101:             'data' => '',
102:             'parameters' => [],
103:             'cookie' => [],
104:             'headers' => [
105:                 'Connection'     =>  'keep-alive',
106:                 'Keep-Alive'     => 300,
107:                 'Accept-Charset' => 'ISO-8859-1,utf-8;q=0.7,*;q=0.7',
108:                 'Accept-Language' => 'en-us,en;q=0.5'           
109:             ],
110:             'options' => [
111:                 'user-agent' => 'Scurl/1.0.3; PHP/' . PHP_VERSION . ' (+http://github.com/sesser/scurl)',
112:                 'timeout' => 10,
113:                 'connect_timeout' => 2,
114:                 'follow_location' => TRUE,
115:                 'max_redirects' => 3
116:             ],
117:         ]);
118:     }
119:     
120:     /**
121:      * Sends the request to the server
122:      * @return \Scurl\Response
123:      * @throws Exceptions\RequestException
124:      */
125:     public function send()
126:     {
127:         if (empty($this->url))
128:             throw new Exceptions\RequestException('No URL has been set for request');
129:         
130:         $ch = curl_init();
131:         $this->setCurlOption($ch, $this->curlopts);
132:         
133:         $query = '';
134:         foreach ($this->parameters as $k => $v)
135:             $query .= (empty($query) ? '?' : '&') . sprintf('%s=%s', $k, urlencode($v));
136:         
137:         if (preg_match('/[\?&]/', $this->url))
138:             $query = substr($query, 1);
139:         
140:         $fp = NULL;
141:         
142:         switch ($this->method)
143:         {
144:             default:
145:             case Request::METHOD_GET:
146:                 $this->setCurlOption($ch, CURLOPT_URL, $this->url . $query);
147:                 break;
148:             case Request::METHOD_POST:
149:                 $this->setCurlOption($ch, [
150:                     CURLOPT_POST => TRUE,
151:                     CURLOPT_POSTFIELDS => $this->parameters,
152:                     CURLOPT_URL => $this->url
153:                 ]);
154:                 break;
155:             case Request::METHOD_HEAD:
156:                 $this->setCurlOption($ch, [
157:                     CURLOPT_NOBODY => TRUE,
158:                     CURLOPT_URL => $this->url . $query
159:                 ]);
160:                 break;
161:             case Request::METHOD_PUT:
162:                 $this->setCurlOption($ch, CURLOPT_URL, $this->url . $query);                
163:                 if (is_file($this->config['data'])) {
164:                     if (FALSE !== ($fp = fopen($this->config['data'], 'r'))) {
165:                         $this->setCurlOption($ch, CURLOPT_PUT, TRUE);
166:                         $this->setCurlOption($ch, CURLOPT_INFILE, $fp);
167:                         $this->setCurlOption($ch, CURLOPT_INFILESIZE, filesize($this->config['data']));
168:                     }
169:                 } else {
170:                     $this->setCurlOption($ch, CURLOPT_CUSTOMREQUEST, Request::METHOD_PUT);
171:                     $this->setCurlOption($ch, CURLOPT_POSTFIELDS, $this->config['data']);
172:                 }
173:                 break;
174:             case Request::METHOD_DELETE:
175:                 $this->setCurlOption($ch, [
176:                     CURLOPT_CUSTOMREQUEST => $this->method,
177:                     CURLOPT_URL => $this->url . $query
178:                 ]);
179:                 break;
180:         }
181:         
182:         $res = curl_exec($ch);
183:         
184:         $errNo = curl_errno($ch);
185:         if ($errNo !== 0)
186:             throw new Exceptions\RequestException(curl_error($ch), $errNo);
187:         
188:         
189:         $response = new Response($res);
190:         $response->info = curl_getinfo($ch);
191:         $response->request_url = $response->info['url'];
192:         $response->request_parameters = $this->parameters;
193:         curl_close($ch);
194:         
195:         return $response;
196:     }
197:     
198:     /**
199:      * Sets a curl option or options if second parameter is an array
200:      * @param resource $ch The cURL resource (from curl_init())
201:      * @param mixed $option A CURLOPT_* option or array of CURLOPT_* => value options
202:      * @param mixed $value The value to set.
203:      * @throws Exceptions\RequestException
204:      */
205:     private function setCurlOption($ch, $option, $value = NULL)
206:     {
207:         if (!is_resource($ch))
208:             throw new Exceptions\RequestException('Parameter #0 is not a valid curl resource');
209:         if (is_array($option)) {
210:             if (function_exists('curl_setopt_array')) {
211:                 curl_setopt_array($ch, $option);
212:             } else {
213:                 foreach ($option as $k => $v)
214:                     $this->setCurlOption($ch, $k, $v);
215:             }
216:         } else {
217:             curl_setopt($ch, $option, $value);
218:         }
219:     }
220:     
221:     /**
222:      * Initializes the request
223:      * @param array $defaults
224:      */
225:     private function initialize(array $defaults)
226:     {
227:         $this->config = Utils::array_merge_recursive($defaults, $this->config);
228:         $this->method = isset($this->config['method']) ? $this->config['method'] : $defaults['method'];
229:         $this->parameters = isset($this->config['parameters']) ? $this->config['parameters'] : [];
230:         $this->options = isset($this->config['options']) ? $this->config['options'] : [];
231:         $this->headers = isset($this->config['headers']) ? $this->config['headers'] : [];
232:         $this->data = isset($this->config['data']) ? $this->config['data'] : NULL;
233:         if (isset($this->config['auth']))
234:             $this->auth = $this->config['auth'] + $defaults['auth'];
235:         
236:         $headers = [];
237:         foreach ($this->headers as $k => $v)
238:             $headers[] = sprintf('%s: %s', $k, $v);
239:         
240:         $this->curlopts = [
241:             CURLOPT_HTTPHEADER => $headers,
242:             CURLOPT_USERAGENT => $this->options['user-agent'],
243:             CURLOPT_HEADER => TRUE,
244:             CURLINFO_HEADER_OUT => TRUE,
245:             CURLOPT_CONNECTTIMEOUT => $this->options['connect_timeout'],
246:             CURLOPT_TIMEOUT => $this->options['timeout'],
247:             CURLOPT_RETURNTRANSFER => TRUE,
248:             CURLOPT_FOLLOWLOCATION => $this->options['follow_location'],
249:             CURLOPT_MAXREDIRS => $this->options['max_redirects'],
250:             CURLOPT_SSL_VERIFYPEER => TRUE,
251:             CURLOPT_SSL_VERIFYHOST => 2,
252:             CURLOPT_CAINFO => __DIR__ . '/cacert.pem',
253:             CURLOPT_ENCODING => ''
254:         ];
255:         
256:         $cookie = '';
257:         if (isset($this->config['cookie'])) {
258:             if (is_array($this->config['cookie'])) {
259:                 $ck = [];
260:                 foreach ($this->config['cookie'] as $k => $v)
261:                     $ck[] = sprintf('%s=%s', $k, $v);
262:                 $cookie = implode('; ', $ck);
263:             }
264:         }
265:         $this->curlopts[CURLOPT_COOKIE] = $cookie;
266:         
267:         if (isset($this->auth['user']) && !empty($this->auth['user'])) {
268:             $this->curlopts[CURLOPT_HTTPAUTH] = CURLAUTH_ANY;
269:             $this->curlopts[CURLOPT_USERPWD] = sprintf('%s:%s', $this->auth['user'], $this->auth['pass']);
270:         }
271:     }
272: 
273: }
