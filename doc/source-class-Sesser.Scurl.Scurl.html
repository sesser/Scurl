 34: /**
 35:  * Scurl
 36:  * 
 37:  * A Simple curl wrapper for making HTTP requests. Supported methods are GET, POST
 38:  * PUT, DELETE and HEAD. With some modifications, it could easily send other
 39:  * custom requests (i.e. OPTIONS). Example:
 40:  * 
 41:  * <code>
 42:  *  $scurl = new Scurl\Scurl(
 43:  *      ['options' => [ 'user-agent' => 'Scurl/1.0; PHP+cURL' ],
 44:  *      ['headers' => [ 'X-Powered-By' => 'My awesome app v0.1a Super Alpha Build' ]
 45:  *  ]);
 46:  *  $response = $scurl->get('http://api.someawesomeservice.com/v1/endpoint', 'param1=value1&foo=bar');
 47:  *  // or the parameters can be an associative array of param => value pairs
 48:  *  $response = $scurl->get('http://api.someawesomeservice.com/v1/endpoint', ['param1' => 'value1', 'foo' => 'bar']);
 49:  *  // or you can pass options to 
 50:  *  $response = $scurl->get('http://api.someawesomeservice.com/v1/endpoint?param1=value1&foo=bar', '', [
 51:  *      'options' => ['user-agent' => 'Private UA']
 52:  *  ]);
 53:  *  $data = json_decode($response->body, TRUE);
 54: 
 55:  * </code>
 56:  * 
 57:  * Supported calls: get, post, put, delete, head
 58:  * 
 59:  * @see Scurl::get
 60:  * @see Scurl::post
 61:  * @see Scurl::put
 62:  * @see Scurl::delete
 63:  * @see Scurl::head
 64:  *
 65:  * @author Randy Sesser <sesser@gmail.com>
 66:  */
 67: class Scurl
 68: {
 69:     /** @var array Configuration */
 70:     protected $config = [];
 71:     
 72:     public function __construct(array $config = [])
 73:     {
 74:         $this->config = $config;
 75:     }
 76:     
 77:     /**
 78:      * Convenience method for GET requests
 79:      * @param string $url The URL to GET
 80:      * @param mixed $params An array of key/value pairs or query string
 81:      * @param array $config Extra config for this request (headers, cookies, etc)
 82:      * @return Response
 83:      */
 84:     public function get($url, $params = '', array $config = [])
 85:     {
 86:         return $this->request($url, $params, $config);
 87:     }
 88:     
 89:     /**
 90:      * Convenience method for POST requests
 91:      * @param string $url The URL in which to POST
 92:      * @param mixed $params An array of key/value pairs or query string
 93:      * @param array $config Extra config for this request (headers, cookies, etc)
 94:      * @return Response
 95:      */
 96:     public function post($url, $params = '', array $config = [])
 97:     {
 98:         return $this->request($url, $params, $config, Request::METHOD_POST);
 99:     }
100:     
101:     /**
102:      * Convenience method for PUT requests
103:      * @param string $url The URL in which to PUT a resource
104:      * @param mixed $params An array of key/value pairs or query string
105:      * @param array $config Extra config for this request (headers, cookies, etc)
106:      * @return Response
107:      */
108:     public function put($url, $params = '', array $config = [])
109:     {
110:         return $this->request($url, $params, $config, Request::METHOD_PUT);
111:     }
112:     
113:     /**
114:      * Convenience method for DELETE requests
115:      * @param string $url The URL to DELETE
116:      * @param mixed $params An array of key/value pairs or query string
117:      * @param array $config Extra config for this request (headers, cookies, etc)
118:      * @return Response
119:      */
120:     public function delete($url, $params = '', array $config = [])
121:     {
122:         return $this->request($url, $params, $config, Request::METHOD_DELETE);
123:     }
124:     
125:     /**
126:      * Convenience method for HEAD requests
127:      * @param string $url The URL
128:      * @param mixed $params An array of key/value pairs or query string
129:      * @param array $config Extra config for this request (headers, cookies, etc)
130:      * @return Response
131:      */
132:     public function head($url, $params = '', array $config = [])
133:     {
134:         return $this->request($url, $params, $config, Request::METHOD_HEAD);
135:     }
136:     
137:     /**
138:      * Creats and sends a new request
139:      * @param string $url The URL. Can contain authentication
140:      * @param mixed $params An associative array of key/value pairs or query string (foo=bar&baz=beer)
141:      * @param array $config Extra configuration for the request (cookies, headers, authentication)
142:      * @param string $method The request method (Request::METHOD_GET, Request::METHOD_POST, etc)
143:      * @return Response
144:      */
145:     public function request($url, $params, array $config = [], $method = Request::METHOD_GET)
146:     {
147:         $config = Utils::array_merge_recursive($this->config, $config);
148:         $config['method'] = $method;
149:         $parameters = [];
150:         if (!is_array($params)) {
151:             parse_str($params, $parameters);
152:         } else {
153:             $parameters = $params;
154:         }
155:         $config['parameters'] = $parameters;
156:         $parsed_url = Utils::array_merge_recursive([
157:             'scheme' => 'http',
158:             'host' => '',
159:             'port' => 80,
160:             'user' => '',
161:             'pass' => '',
162:             'path' => '',
163:             'query' => '',
164:             'fragment' => ''
165:         ], parse_url($url));
166:         
167:         extract($parsed_url);
168:         
169:         if (!empty($user)) 
170:             $config['auth'] = [ 'user' => $user, 'pass' => $pass ];
171:         
172:         $url = Utils::http_build_url($parsed_url);
173:         
174:         $request = new Request($url, $config);
175:         return $request->send();        
176:     }
177: }
